<script>
    $(document).ready(function(){
        $('#slideout-close').click(function(){
            $(this).closest('.new-order-slideout').animate({
                    'right':'-186px'
                },function(){
                    $(this).hide();
                });
        });
        $('#new-observation').click(function(){
            var slideOut = $(this).siblings('.new-order-slideout');
            if (slideOut.is(':hidden')){
                slideOut.show();
                slideOut.animate({
                    'right':'0'
                });
                slideOut.focus();
            } else {
                slideOut.animate({
                    'right':'-186px'
                },function(){
                    $(this).hide();
                });
            }
        });
    })
</script>
